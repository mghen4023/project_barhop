Database SQL

--

-- Database: `barhop_dev`

-- --------------------------------------------------------

 

--

-- Table structure for table `drink`

--

 

CREATE TABLE IF NOT EXISTS `drink` (

  `drink_id` int(11) NOT NULL COMMENT 'The primary key ',

  `name` varchar(100) NOT NULL COMMENT 'The human readable name of this drink',

  `description` text NOT NULL COMMENT 'An optional description about this drink',

  `price` decimal(6,0) NOT NULL COMMENT 'The price in dollars of a drink',

  `location_id` int(11) NOT NULL COMMENT 'The id of the locaiton this drink is at',

  PRIMARY KEY (`drink_id`)

) ENGINE=InnoDB DEFAULT CHARSET=latin1 COMMENT='This tables tracks data relating to drinks. Each entry is a drink at a locaiton';

 

-- --------------------------------------------------------

 

--

-- Table structure for table `drink_special`

--

 

CREATE TABLE IF NOT EXISTS `drink_special` (

  `ds_id` int(11) NOT NULL AUTO_INCREMENT COMMENT 'The primary key',

  `special` varchar(255) NOT NULL COMMENT 'The human readable name of this speical',

  `description` text NOT NULL COMMENT 'An optional description of this special',

  `date` date NOT NULL COMMENT 'The date this special is valid on',

  `time` time NOT NULL COMMENT 'The time that this special is valid',

  `discount` decimal(10,0) NOT NULL COMMENT 'A % of $ discount',

  `location_id` int(11) NOT NULL COMMENT 'The id of the location of this drink speical',

  PRIMARY KEY (`ds_id`)

) ENGINE=InnoDB DEFAULT CHARSET=latin1 COMMENT='This table holds information about the drink specials being offered.' AUTO_INCREMENT=1 ;

 

-- --------------------------------------------------------

 

--

-- Table structure for table `food`

--

 

CREATE TABLE IF NOT EXISTS `food` (

  `food_id` int(11) NOT NULL AUTO_INCREMENT COMMENT 'The primary key',

  `name` varchar(200) NOT NULL COMMENT 'The name of this food item',

  `description` text NOT NULL COMMENT 'A optional description of this food ',

  `price` decimal(6,0) NOT NULL COMMENT 'The price in dollars of this food',

  `location_id` int(11) NOT NULL COMMENT 'The id of the location this food is served at',

  PRIMARY KEY (`food_id`)

) ENGINE=InnoDB DEFAULT CHARSET=latin1 COMMENT='The food table holds informtion about food items from various locations' AUTO_INCREMENT=1 ;

 

-- --------------------------------------------------------

 

--

-- Table structure for table `food_special`

--

 

CREATE TABLE IF NOT EXISTS `food_special` (
